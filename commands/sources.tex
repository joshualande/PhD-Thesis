% IfEqCase comes from xstring.
% The DeclareRobustCommand is somewhat criptic, but
% has something to do with the fact that IfEqCase
% is fragile and does not work well in section titles.
% See http://tex.stackexchange.com/questions/4736/what-is-the-difference-between-fragile-and-robust-commands
% Or http://tex.stackexchange.com/questions/61503/newcommand-vs-declarerobustcommand
\newcommand{\basehess}[3]{HESS$\,$J#1$#2$#3\xspace}
\DeclareRobustCommand{\hess}[1]{
\IfEqCase{#1}{
{J1825}{\basehess{1825}{-}{137}}
{J1640}{\basehess{1640}{-}{465}}
}[\PackageError{hess}{Undefined option to hess#1}{}]
}

\newcommand{\basetwofgl}[3]{2FGL$\,$J#1$#2$#3\xspace}
\DeclareRobustCommand{\twofgl}[1]{
\IfEqCase{#1}{
{J1857}{\basetwofgl{1857}{+}{026}}
{X}{X}
}[\PackageError{hess}{Undefined option to hess#1}{}]
}
